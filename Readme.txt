安装webpack和webpack的命令行工具
npm install --save-dev webpack webpack-cli

安装本地服务器
npm install --save-dev webpack webpack-dev-server

npm install --save-dev babel-core babel-loader url-loader file-loader
npm install --save-dev babel-preset-es2015 babel-preset-react babel-preset-react-native babel-plugin-transform-react-remove-prop-types

安装react-native-web
npm install --save react react-dom react-native-web
